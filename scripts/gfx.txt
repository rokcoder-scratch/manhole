{
    List.Add(DEBUG, comment);
    List.Show(DEBUG);
}

Define process graphics queue (warp=true)
{
    i = 1;
    Repeat Until (i > List.Length(Gfx.queue))
    {
        command = Gfx.queue[i];
        i += 1;
        If (command == command.clearScr)
        {
            Pen.Clear();
        }
        Else
        {
            If (command == command.drawPic)
            {
                Looks.SwitchCostumeTo(Operator.Join("flex-", Gfx.queue[i]));
                i += 1;
                Motion.SetX(((Gfx.queue[i] + (costume dimensions[((Costume.GetNumberName(number) * 2) - 1)] / 2)) - 160));
                i += 1;
                Motion.SetY(((100 - Gfx.queue[i]) - (costume dimensions[(Costume.GetNumberName(number) * 2)] / 2)));
                i += 1;
                flipX = Gfx.queue[i];
                i += 1;
                flipY = Gfx.queue[i];
                i += 1;
                If (Not (((flipX == 0) And (flipY == 0))))
                {
                    Call //TODO(Operator.Join(Operator.Join(Operator.Join(Operator.Join("flipX and/or flipY need implementing (flipX = ", flipX), ", flipY = "), flipY), ")"));
                }
                Pen.Stamp();
            }
            Else
            {
            }
        }
    }
    List.DeleteAll(Gfx.queue);
}
