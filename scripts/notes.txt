//----- Variables -----------------------------------------------------------------------------------------------------

clone? = 0
instrument = 6
note = 64



//----- Broadcast received events -------------------------------------------------------------------------------------

WhenBroadcastReceived(initialise notes)
{
    clone? = False;
}

WhenBroadcastReceived(play note)
{
    If (clone? == False)
    {
        note = (BroadcastParam1 - 1);
        instrument = BroadcastParam2;
        Control.CreateCloneOf(_myself_);
    }
}

WhenBroadcastReceived(stop note)
{
    If (clone? == True)
    {
        Control.DeleteThisClone();
    }
}


//----- Start as clone ------------------------------------------------------------------------------------------------

Control.WhenIStartAsClone()
{
    clone? = True;
    Music.SetInstrument(instrument);
    Music.PlayNote(note, 100);
    Control.DeleteThisClone();
}
