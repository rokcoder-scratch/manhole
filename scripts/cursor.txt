//----- Variables -----------------------------------------------------------------------------------------------------

offsetX = 0
offsetY = 0
visible = 1



//----- Broadcast received events -------------------------------------------------------------------------------------

WhenBroadcastReceived(hide cursor)
{
    visible = False;
    Looks.Hide();
}

WhenBroadcastReceived(initialise cursor)
{
    offsetX = 0;
    offsetY = 0;
    visible = False;
    Looks.SetSizeTo(300);
}

WhenBroadcastReceived(MadeEngine::handleEvents)
{
    If (visible == True)
    {
        Motion.GoToXY((Sensing.MouseX() + offsetX), (Sensing.MouseY() + offsetY));
    }
}

WhenBroadcastReceived(show cursor)
{
    visible = True;
    Looks.Show();
}

WhenBroadcastReceived(thumbnail)
{
    Looks.Hide();
}
